				add_claim = ROOT
			}
		}

			limit = {
				exists = scope:SAO_PAULO_owner
			}
			create_incident = {
				country = scope:SAO_PAULO_owner
				value = 3
			}
		}
	}

		name = minas.2.b
		default_option = yes
		trigger = {
			OR = {
				NOT = { owns_entire_state_region = STATE_PARANA }
				NOT = { owns_entire_state_region = STATE_GOIAS }
			}
		}
		if = {
