je_mineiros_ambition = {
	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	is_shown_when_inactive = {
		country_has_primary_culture = cu:Mineiro
		any_scope_state = {
			state_region = {
				is_homeland = cu:Mineiro
			}
			count >= 7
		}
	}

	possible = {
		has_technology_researched = nationalism
		custom_tooltip = {
			text = mineiros_state_possible_2_tt
			any_scope_state = {
				state_region = {
					is_homeland = cu:Mineiro
				}
				count >= 7
			}
		}
		is_subject = no
	}

	immediate = {
		trigger_event = { id = minas.3 }
	}

	complete = {
		OR = {
			AND = { 
				owns_entire_state_region = STATE_GOIAS
				custom_tooltip = {
					text = mineiros_state_complete_2_tt
					any_scope_state = {
						state_region = {
							is_homeland = cu:Mineiro
						}
						count >= 8
					}
				}
			}
			AND = { 
				has_law = law_type:law_autocracy
				OR = {
					has_law = law_type:law_ethnostate
					has_law = law_type:law_national_supremacy
				}
			}
		}
	}

	on_complete = {
		trigger_event = { id = minas.4 }
	}
	
	should_be_pinned_by_default = yes
	
}

			}
		}
	}

		trigger_event = { id = minas.8 }
	}
